# Zagon skripte
sbatch spark-slurm.sh

# Preverba stanja
sacct -j JOB_ID

# Prikaz svojih nalog
squeue -u campus02

# Prikaz vseh nalog in podrobnosti/ene naloge
scontrol show job [JOB_ID]

# Preklic izvajanja
scancel JOB_ID
